Shared Temp Tables (prepend to segment queries)

⚠️ Must run in the same session as the query that uses them Used by: Adoption, Product Mix, Txn Rate queries
These temp tables deduplicate catalog SKUs to the latest event per (sku_id, store_id) and build a store → business UUID mapping via the latest snapshot.
-- Latest catalog SKU snapshot per store
DROP TABLE IF EXISTS catalog_sku_job_cln;
CREATE TEMP TABLE catalog_sku_job_cln AS
SELECT a.*
FROM dev.signals_platform_cln.catalog_sku_job_cln a
INNER JOIN (
    SELECT sku_id, store_id, MAX(event_utc_ts) AS max_ts
    FROM dev.signals_platform_cln.catalog_sku_job_cln
    WHERE created_at_utc_ts >= '2025-01-01'
    GROUP BY 1, 2
) b ON a.sku_id = b.sku_id
    AND a.store_id = b.store_id
    AND a.event_utc_ts = b.max_ts
WHERE a.created_at_utc_ts >= '2025-01-01';

-- Store UUID → Business UUID mapping (latest snapshot, US only)
DROP TABLE IF EXISTS store_business_map;
CREATE TEMP TABLE store_business_map AS
SELECT s.store_uuid, b.business_uuid
FROM poynt_cln.business_cln b
JOIN poynt_cln.store_cln s
    ON b.business_id = s.business_id AND s.region = b.region
WHERE b.snap_date = (SELECT MAX(snap_date) FROM poynt_cln.business_cln WHERE region = 'US')
  AND s.snap_date = (SELECT MAX(snap_date) FROM poynt_cln.store_cln WHERE region = 'US');

30-Day and 90-Day Catalog Product Adoption % by KYC Month

📊 Charts 1 & 2 Tables: dna_approved.payment_360, catalog_sku_job_cln (temp), store_business_map (temp) 36 rows
Prepend the shared temp table SQL above before running. Adoption = at least one catalog product added (SKU created) within the window, not a transaction.
WITH merchant_kyc AS (
    SELECT
        business_uuid,
        current_acquisition_channel,
        full_application_approval_utc_ts            AS kyc_ts,
        full_application_approval_utc_ts::date      AS kyc_date,
        DATE_TRUNC('month', full_application_approval_utc_ts) AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL
      AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-01-01'
      AND full_application_approval_utc_ts < '2026-07-01'
      AND current_acquisition_channel IN ('Sales Initiated', 'Self Serve')
),
adopters_30d AS (
    SELECT DISTINCT m.business_uuid, m.current_acquisition_channel, m.kyc_month
    FROM merchant_kyc m
    JOIN store_business_map sbm ON m.business_uuid = sbm.business_uuid
    JOIN catalog_sku_job_cln c  ON sbm.store_uuid = c.store_id
        AND c.created_at_utc_ts >= m.kyc_ts
        AND c.created_at_utc_ts <= DATEADD(day, 30, m.kyc_date)
),
adopters_90d AS (
    SELECT DISTINCT m.business_uuid, m.current_acquisition_channel, m.kyc_month
    FROM merchant_kyc m
    JOIN store_business_map sbm ON m.business_uuid = sbm.business_uuid
    JOIN catalog_sku_job_cln c  ON sbm.store_uuid = c.store_id
        AND c.created_at_utc_ts >= m.kyc_ts
        AND c.created_at_utc_ts <= DATEADD(day, 90, m.kyc_date)
),
cohort AS (
    SELECT current_acquisition_channel, kyc_month,
        COUNT(DISTINCT business_uuid) AS total_kyc_merchants
    FROM merchant_kyc
    GROUP BY 1, 2
)
SELECT
    c.kyc_month,
    c.current_acquisition_channel,
    c.total_kyc_merchants,
    COUNT(DISTINCT a30.business_uuid)                                       AS catalog_adopters_30d,
    ROUND(100.0 * COUNT(DISTINCT a30.business_uuid)
          / NULLIF(c.total_kyc_merchants, 0), 2)                            AS catalog_adoption_30d_pct,
    COUNT(DISTINCT a90.business_uuid)                                       AS catalog_adopters_90d,
    ROUND(100.0 * COUNT(DISTINCT a90.business_uuid)
          / NULLIF(c.total_kyc_merchants, 0), 2)                            AS catalog_adoption_90d_pct
FROM cohort c
LEFT JOIN adopters_30d a30
    ON c.kyc_month = a30.kyc_month
   AND c.current_acquisition_channel = a30.current_acquisition_channel
LEFT JOIN adopters_90d a90
    ON c.kyc_month = a90.kyc_month
   AND c.current_acquisition_channel = a90.current_acquisition_channel
GROUP BY 1, 2, 3
ORDER BY 1, 2;

30-Day Catalog Adopters: % with No Catalog Txn vs No Any Txn in 90 Days

📊 Chart 3 Tables: dna_approved.payment_360, catalog_sku_job_cln (temp), store_business_map (temp), dev.ba_commerce.txn_w_catalog 72 rows (long format — 4 series)
Prepend the shared temp table SQL above before running. Solid lines = no catalog transactions in 90 days. Dashed lines = no transactions of any kind in 90 days. The gap between the two lines per channel reveals merchants who ARE transacting via other channels but not through catalog.
WITH merchant_kyc AS (
    SELECT
        business_uuid,
        current_acquisition_channel,
        full_application_approval_utc_ts            AS kyc_ts,
        full_application_approval_utc_ts::date      AS kyc_date,
        DATE_TRUNC('month', full_application_approval_utc_ts) AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL
      AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-01-01'
      AND full_application_approval_utc_ts < '2026-07-01'
      AND current_acquisition_channel IN ('Sales Initiated', 'Self Serve')
),
adopters_30d AS (
    SELECT DISTINCT m.business_uuid, m.current_acquisition_channel, m.kyc_month,
        m.kyc_ts, m.kyc_date
    FROM merchant_kyc m
    JOIN store_business_map sbm ON m.business_uuid = sbm.business_uuid
    JOIN catalog_sku_job_cln c  ON sbm.store_uuid = c.store_id
        AND c.created_at_utc_ts >= m.kyc_ts
        AND c.created_at_utc_ts <= DATEADD(day, 30, m.kyc_date)
),
catalog_txn_90d AS (
    SELECT DISTINCT a.business_uuid, a.kyc_month, a.current_acquisition_channel
    FROM adopters_30d a
    JOIN dev.ba_commerce.txn_w_catalog t ON t.business_uuid = a.business_uuid
    WHERE t.catalog_item_flag = 1
      AND t.create_utc_ts >= a.kyc_ts
      AND t.create_utc_ts <= DATEADD(day, 90, a.kyc_date)
),
any_txn_90d AS (
    SELECT DISTINCT a.business_uuid, a.kyc_month, a.current_acquisition_channel
    FROM adopters_30d a
    JOIN dev.ba_commerce.txn_w_catalog t ON t.business_uuid = a.business_uuid
    WHERE t.create_utc_ts >= a.kyc_ts
      AND t.create_utc_ts <= DATEADD(day, 90, a.kyc_date)
),
base AS (
    SELECT
        a.kyc_month,
        a.current_acquisition_channel,
        COUNT(DISTINCT a.business_uuid)          AS adopters_30d,
        COUNT(DISTINCT ct.business_uuid)         AS with_catalog_txn_90d,
        COUNT(DISTINCT at.business_uuid)         AS with_any_txn_90d
    FROM adopters_30d a
    LEFT JOIN catalog_txn_90d ct
        ON a.business_uuid = ct.business_uuid
       AND a.kyc_month = ct.kyc_month
       AND a.current_acquisition_channel = ct.current_acquisition_channel
    LEFT JOIN any_txn_90d at
        ON a.business_uuid = at.business_uuid
       AND a.kyc_month = at.kyc_month
       AND a.current_acquisition_channel = at.current_acquisition_channel
    GROUP BY 1, 2
)
SELECT kyc_month,
    current_acquisition_channel || ' — No Catalog Txn' AS series,
    ROUND(100.0 * (adopters_30d - with_catalog_txn_90d) / NULLIF(adopters_30d, 0), 2) AS pct
FROM base
UNION ALL
SELECT kyc_month,
    current_acquisition_channel || ' — No Any Txn' AS series,
    ROUND(100.0 * (adopters_30d - with_any_txn_90d) / NULLIF(adopters_30d, 0), 2) AS pct
FROM base
ORDER BY 1, 2;

30-Day Product Count Mix by Segment — Sales Initiated / Self Serve

📊 Charts 4 & 5 Tables: dna_approved.payment_360, catalog_sku_job_cln (temp), store_business_map (temp) 90 rows each
Prepend the shared temp table SQL above before running. Run once per channel by changing the filter below (Chart 4 = Sales Initiated, Chart 5 = Self Serve).
WITH merchant_kyc AS (
    SELECT
        business_uuid,
        full_application_approval_utc_ts,
        full_application_approval_utc_ts::date      AS kyc_date,
        DATE_TRUNC('month', full_application_approval_utc_ts) AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL
      AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-01-01'
      AND full_application_approval_utc_ts < '2026-07-01'
      AND current_acquisition_channel = 'Sales Initiated'   -- change to 'Self Serve' for Chart 5
),
products AS (
    SELECT m.business_uuid, COUNT(DISTINCT c.sku_id) AS n
    FROM merchant_kyc m
    JOIN store_business_map sbm ON m.business_uuid = sbm.business_uuid
    JOIN catalog_sku_job_cln c  ON sbm.store_uuid = c.store_id
        AND c.created_at_utc_ts >= m.full_application_approval_utc_ts
        AND c.created_at_utc_ts <= DATEADD(day, 30, m.kyc_date)
    GROUP BY 1
),
seg AS (
    SELECT
        m.kyc_month,
        m.business_uuid,
        CASE
            WHEN p.n >= 50 THEN '50+'
            WHEN p.n >= 25 THEN '25+'
            WHEN p.n >= 10 THEN '10+'
            WHEN p.n >= 5  THEN '5+'
            ELSE '1+'
        END AS product_segment
    FROM merchant_kyc m
    JOIN products p ON m.business_uuid = p.business_uuid
    WHERE p.n >= 1
)
SELECT
    kyc_month,
    product_segment,
    COUNT(DISTINCT business_uuid)                                          AS merchant_count,
    ROUND(100.0 * COUNT(DISTINCT business_uuid)
          / NULLIF(SUM(COUNT(DISTINCT business_uuid)) OVER (PARTITION BY kyc_month), 0), 2) AS pct_of_channel
FROM seg
GROUP BY 1, 2
ORDER BY 1, 2;

30-Day and 90-Day Catalog Txn Rate by Product Segment — Sales Initiated / Self Serve

📊 Charts 6–9 Tables: dna_approved.payment_360, catalog_sku_job_cln (temp), store_business_map (temp), dev.ba_commerce.txn_w_catalog 90 rows each
Prepend the shared temp table SQL above before running. Chart 6 = SI/30-day, Chart 7 = SS/30-day, Chart 8 = SI/90-day, Chart 9 = SS/90-day — toggle the channel filter and the two DATEADD windows (30 or 90) together.
WITH merchant_kyc AS (
    SELECT
        business_uuid,
        full_application_approval_utc_ts,
        full_application_approval_utc_ts::date      AS kyc_date,
        DATE_TRUNC('month', full_application_approval_utc_ts) AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL
      AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-01-01'
      AND full_application_approval_utc_ts < '2026-07-01'
      AND current_acquisition_channel = 'Sales Initiated'   -- change to 'Self Serve' for SS
),
products AS (
    SELECT m.business_uuid, COUNT(DISTINCT c.sku_id) AS n
    FROM merchant_kyc m
    JOIN store_business_map sbm ON m.business_uuid = sbm.business_uuid
    JOIN catalog_sku_job_cln c  ON sbm.store_uuid = c.store_id
        AND c.created_at_utc_ts >= m.full_application_approval_utc_ts
        AND c.created_at_utc_ts <= DATEADD(day, 30, m.kyc_date)  -- 30 or 90
    GROUP BY 1
),
seg AS (
    SELECT
        m.kyc_month,
        m.business_uuid,
        m.full_application_approval_utc_ts,
        CASE
            WHEN p.n >= 50 THEN '50+'
            WHEN p.n >= 25 THEN '25+'
            WHEN p.n >= 10 THEN '10+'
            WHEN p.n >= 5  THEN '5+'
            ELSE '1+'
        END AS product_segment
    FROM merchant_kyc m
    JOIN products p ON m.business_uuid = p.business_uuid
    WHERE p.n >= 1
),
txn AS (
    SELECT DISTINCT t.business_uuid
    FROM dev.ba_commerce.txn_w_catalog t
    JOIN seg s ON t.business_uuid = s.business_uuid
    WHERE t.catalog_item_flag = 1
      AND t.create_utc_ts >= s.full_application_approval_utc_ts
      AND t.create_utc_ts <= DATEADD(day, 30, s.full_application_approval_utc_ts::date)  -- 30 or 90
)
SELECT
    s.kyc_month,
    s.product_segment,
    COUNT(DISTINCT s.business_uuid)                                        AS segment_merchants,
    COUNT(DISTINCT t.business_uuid)                                        AS catalog_txn_merchants,
    ROUND(100.0 * COUNT(DISTINCT t.business_uuid)
          / NULLIF(COUNT(DISTINCT s.business_uuid), 0), 2)                 AS catalog_txn_pct
FROM seg s
LEFT JOIN txn t ON s.business_uuid = t.business_uuid
GROUP BY 1, 2
ORDER BY 1, 2;

SI: 30-Day and 90-Day Catalog Usage % by Signed aGPV Band

📊 Charts 10 & 11 Tables: dna_approved.payment_360, ba_commerce.final_campaign_table, dev.ba_commerce.txn_w_catalog 126 rows (30d) / 105 rows (90d)
Signed aGPV comes from the sales opportunity record (gd_sales_estimate_agpv_usd_amt), joined on gd_merchant_id = business_uuid. "Usage" here = had at least one catalog transaction within the window, not just a product add. For Chart 11 (90-day), change the window to 90 and restrict full_application_approval_utc_ts < '2026-04-01' so every cohort has a complete 90-day observation period.
WITH merchant_kyc AS (
    SELECT business_uuid,
        full_application_approval_utc_ts::date AS kyc_date,
        TO_CHAR(DATE_TRUNC('month', full_application_approval_utc_ts), 'YYYY-MM') AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL
      AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-01-01'
      AND full_application_approval_utc_ts < '2026-07-01'   -- use '2026-04-01' for 90-day chart
      AND current_acquisition_channel = 'Sales Initiated'
),
campaigns AS (
    SELECT gd_merchant_id,
        MAX(gd_sales_estimate_agpv_usd_amt) AS signed_agpv
    FROM ba_commerce.final_campaign_table
    WHERE gd_merchant_id IS NOT NULL
    GROUP BY 1
),
merchant_agpv AS (
    SELECT m.business_uuid, m.kyc_date, m.kyc_month,
        CASE
            WHEN c.signed_agpv < 50000                                THEN '1. Under $50K'
            WHEN c.signed_agpv >= 50000  AND c.signed_agpv < 100000  THEN '2. $50K–$100K'
            WHEN c.signed_agpv >= 100000 AND c.signed_agpv < 250000  THEN '3. $100K–$250K'
            WHEN c.signed_agpv >= 250000 AND c.signed_agpv < 500000  THEN '4. $250K–$500K'
            WHEN c.signed_agpv >= 500000 AND c.signed_agpv < 1000000 THEN '5. $500K–$1M'
            WHEN c.signed_agpv >= 1000000                            THEN '6. Over $1M'
            ELSE 'Unknown'
        END AS signed_agpv_band
    FROM merchant_kyc m
    LEFT JOIN campaigns c ON m.business_uuid = c.gd_merchant_id
),
catalog_txn AS (
    SELECT t.business_uuid,
        MIN(t.create_utc_ts::date) AS first_catalog_txn_date
    FROM dev.ba_commerce.txn_w_catalog t
    WHERE t.catalog_item_flag = 1
      AND t.create_utc_ts >= '2025-01-01'
    GROUP BY 1
)
SELECT
    m.signed_agpv_band,
    m.kyc_month,
    COUNT(DISTINCT m.business_uuid) AS total_merchants,
    ROUND(100.0 * COUNT(DISTINCT CASE WHEN ct.first_catalog_txn_date <= m.kyc_date + 30   -- or +90
                                      THEN m.business_uuid END)
          / NULLIF(COUNT(DISTINCT m.business_uuid), 0), 2) AS catalog_pct_30d   -- or catalog_pct_90d
FROM merchant_agpv m
LEFT JOIN catalog_txn ct ON m.business_uuid = ct.business_uuid
GROUP BY 1, 2
ORDER BY 1, 2;

Catalog Drop-Off Distribution (Last Active Month)

📊 Chart 12 Tables: dna_approved.payment_360, dev.ba_commerce.txn_w_catalog 38 rows 
WITH merchant_kyc AS (
    SELECT
        business_uuid,
        current_acquisition_channel,
        full_application_approval_utc_ts::date      AS kyc_date,
        DATE_TRUNC('month', full_application_approval_utc_ts) AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL
      AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-01-01'
      AND full_application_approval_utc_ts < '2026-07-01'
      AND current_acquisition_channel IN ('Sales Initiated', 'Self Serve')
),
last_active AS (
    SELECT
        m.business_uuid,
        m.current_acquisition_channel,
        DATEDIFF('month', m.kyc_month,
                 DATE_TRUNC('month', MAX(t.create_utc_ts))) AS last_active_month
    FROM merchant_kyc m
    JOIN dev.ba_commerce.txn_w_catalog t ON t.business_uuid = m.business_uuid
    WHERE t.catalog_item_flag = 1
      AND t.create_utc_ts >= m.kyc_date
      AND t.create_utc_ts >= '2025-01-01'
    GROUP BY 1, 2, m.kyc_month
),
cohort AS (
    SELECT current_acquisition_channel,
        COUNT(DISTINCT business_uuid) AS total
    FROM last_active GROUP BY 1
)
SELECT
    la.current_acquisition_channel,
    la.last_active_month,
    COUNT(DISTINCT la.business_uuid)                                       AS merchant_count,
    ROUND(100.0 * COUNT(DISTINCT la.business_uuid)
          / NULLIF(c.total, 0), 2)                                         AS pct_of_channel
FROM last_active la
JOIN cohort c ON la.current_acquisition_channel = c.current_acquisition_channel
WHERE la.last_active_month >= 0
GROUP BY 1, 2
ORDER BY 1, 2;

All-Txn vs Catalog-Txn Drop-Off Comparison (4 Series)

📊 Chart 13 Tables: dna_approved.payment_360, dev.ba_commerce.txn_w_catalog 76 rows 
WITH merchant_kyc AS (
    SELECT
        business_uuid,
        current_acquisition_channel,
        full_application_approval_utc_ts::date      AS kyc_date,
        DATE_TRUNC('month', full_application_approval_utc_ts) AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL
      AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-01-01'
      AND full_application_approval_utc_ts < '2026-07-01'
      AND current_acquisition_channel IN ('Sales Initiated', 'Self Serve')
),
catalog_last AS (
    SELECT m.business_uuid, m.current_acquisition_channel,
        DATEDIFF('month', m.kyc_month, DATE_TRUNC('month', MAX(t.create_utc_ts))) AS last_active_month
    FROM merchant_kyc m
    JOIN dev.ba_commerce.txn_w_catalog t ON t.business_uuid = m.business_uuid
    WHERE t.catalog_item_flag = 1 AND t.create_utc_ts >= m.kyc_date AND t.create_utc_ts >= '2025-01-01'
    GROUP BY 1, 2, m.kyc_month
),
all_last AS (
    SELECT m.business_uuid, m.current_acquisition_channel,
        DATEDIFF('month', m.kyc_month, DATE_TRUNC('month', MAX(t.create_utc_ts))) AS last_active_month
    FROM merchant_kyc m
    JOIN dev.ba_commerce.txn_w_catalog t ON t.business_uuid = m.business_uuid
    WHERE t.create_utc_ts >= m.kyc_date AND t.create_utc_ts >= '2025-01-01'
    GROUP BY 1, 2, m.kyc_month
),
catalog_cohort AS (SELECT current_acquisition_channel, COUNT(DISTINCT business_uuid) AS total FROM catalog_last GROUP BY 1),
all_cohort    AS (SELECT current_acquisition_channel, COUNT(DISTINCT business_uuid) AS total FROM all_last    GROUP BY 1),
catalog_dist  AS (SELECT current_acquisition_channel, last_active_month, COUNT(DISTINCT business_uuid) AS cnt FROM catalog_last WHERE last_active_month >= 0 GROUP BY 1, 2),
all_dist      AS (SELECT current_acquisition_channel, last_active_month, COUNT(DISTINCT business_uuid) AS cnt FROM all_last    WHERE last_active_month >= 0 GROUP BY 1, 2)
SELECT 'Catalog Txn' AS txn_type, d.current_acquisition_channel, d.last_active_month,
    ROUND(100.0 * d.cnt / NULLIF(c.total, 0), 2) AS pct_of_cohort
FROM catalog_dist d JOIN catalog_cohort c ON d.current_acquisition_channel = c.current_acquisition_channel
UNION ALL
SELECT 'All Txn' AS txn_type, d.current_acquisition_channel, d.last_active_month,
    ROUND(100.0 * d.cnt / NULLIF(c.total, 0), 2) AS pct_of_cohort
FROM all_dist d JOIN all_cohort c ON d.current_acquisition_channel = c.current_acquisition_channel
ORDER BY 1, 2, 3;

All-Txn vs Catalog-Txn Retention Comparison (4 Series)

📊 Chart 14 Tables: dna_approved.payment_360, dev.ba_commerce.txn_w_catalog 76 rows 
WITH merchant_kyc AS (
    SELECT
        business_uuid,
        current_acquisition_channel,
        full_application_approval_utc_ts::date      AS kyc_date,
        DATE_TRUNC('month', full_application_approval_utc_ts) AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL
      AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-01-01'
      AND full_application_approval_utc_ts < '2026-07-01'
      AND current_acquisition_channel IN ('Sales Initiated', 'Self Serve')
),
-- Catalog-transaction retention
cat_txn AS (
    SELECT m.business_uuid, m.current_acquisition_channel,
        DATEDIFF('month', m.kyc_month, DATE_TRUNC('month', t.create_utc_ts)) AS months_since_kyc,
        'Catalog Txn' AS txn_type
    FROM merchant_kyc m
    JOIN dev.ba_commerce.txn_w_catalog t ON t.business_uuid = m.business_uuid
    WHERE t.catalog_item_flag = 1 AND t.create_utc_ts >= m.kyc_date AND t.create_utc_ts >= '2025-01-01'
    GROUP BY 1, 2, 3, 4
),
-- All-transaction retention
all_txn AS (
    SELECT m.business_uuid, m.current_acquisition_channel,
        DATEDIFF('month', m.kyc_month, DATE_TRUNC('month', t.create_utc_ts)) AS months_since_kyc,
        'All Txn' AS txn_type
    FROM merchant_kyc m
    JOIN dev.ba_commerce.txn_w_catalog t ON t.business_uuid = m.business_uuid
    WHERE t.create_utc_ts >= m.kyc_date AND t.create_utc_ts >= '2025-01-01'
    GROUP BY 1, 2, 3, 4
),
combined AS (SELECT * FROM cat_txn UNION ALL SELECT * FROM all_txn),
cohort AS (
    SELECT txn_type, current_acquisition_channel, COUNT(DISTINCT business_uuid) AS total
    FROM combined GROUP BY 1, 2
),
ret AS (
    SELECT txn_type, current_acquisition_channel, months_since_kyc,
        COUNT(DISTINCT business_uuid) AS active
    FROM combined WHERE months_since_kyc >= 0 GROUP BY 1, 2, 3
)
SELECT
    r.current_acquisition_channel || ' — ' || r.txn_type AS series,
    r.months_since_kyc,
    ROUND(100.0 * r.active / NULLIF(c.total, 0), 2) AS retention_pct
FROM ret r
JOIN cohort c ON r.txn_type = c.txn_type
    AND r.current_acquisition_channel = c.current_acquisition_channel
ORDER BY 1, 2;

SI — Catalog Retention by Cohort Month

📊 Chart 15 Tables: dna_approved.payment_360, dev.ba_commerce.txn_w_catalog 90 rows
Denominator per cohort month = merchants from that KYC month who ever had a catalog transaction post-KYC. Restricted to Jul 2025–Jun 2026 (last 12 months) so the chart stays readable — widen the date filter for the full history. Each cohort month becomes one line at x = months_since_kyc.
WITH merchant_kyc AS (
    SELECT business_uuid,
        full_application_approval_utc_ts::date AS kyc_date,
        DATE_TRUNC('month', full_application_approval_utc_ts) AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-07-01' AND full_application_approval_utc_ts < '2026-07-01'
      AND current_acquisition_channel = 'Sales Initiated'
),
txn AS (
    SELECT m.business_uuid, m.kyc_month,
        DATEDIFF('month', m.kyc_month, DATE_TRUNC('month', t.create_utc_ts)) AS months_since_kyc
    FROM merchant_kyc m
    JOIN dev.ba_commerce.txn_w_catalog t ON t.business_uuid = m.business_uuid
    WHERE t.catalog_item_flag = 1
      AND t.create_utc_ts >= m.kyc_date
      AND t.create_utc_ts >= '2025-07-01'
    GROUP BY 1, 2, 3
),
cohort AS (
    SELECT kyc_month, COUNT(DISTINCT business_uuid) AS total
    FROM txn GROUP BY 1
),
ret AS (
    SELECT kyc_month, months_since_kyc, COUNT(DISTINCT business_uuid) AS active
    FROM txn WHERE months_since_kyc >= 0 GROUP BY 1, 2
)
SELECT
    TO_CHAR(r.kyc_month, 'YYYY-MM') AS kyc_month,
    r.months_since_kyc,
    c.total AS cohort_size,
    ROUND(100.0 * r.active / NULLIF(c.total, 0), 2) AS retention_pct
FROM ret r
JOIN cohort c ON r.kyc_month = c.kyc_month
ORDER BY 1, 2;

SI — Catalog Retention by Cohort Month & Signed aGPV Band

📊 Chart 16 Tables: dna_approved.payment_360, ba_commerce.final_campaign_table, dev.ba_commerce.txn_w_catalog 600 rows
Same cohort-retention logic as Chart 15, but the "ever active" cohort and retention denominator are also split by signed aGPV band. The dashboard renders this as a dropdown that filters to one band at a time (one line per cohort month within the selected band).
WITH merchant_kyc AS (
    SELECT business_uuid,
        full_application_approval_utc_ts::date AS kyc_date,
        DATE_TRUNC('month', full_application_approval_utc_ts) AS kyc_month
    FROM dna_approved.payment_360
    WHERE full_application_approval_utc_ts IS NOT NULL AND type_desc_text != 'TEST_MERCHANT'
      AND full_application_approval_utc_ts >= '2025-07-01' AND full_application_approval_utc_ts < '2026-07-01'
      AND current_acquisition_channel = 'Sales Initiated'
),
campaigns AS (
    SELECT gd_merchant_id, MAX(gd_sales_estimate_agpv_usd_amt) AS signed_agpv
    FROM ba_commerce.final_campaign_table
    WHERE gd_merchant_id IS NOT NULL GROUP BY 1
),
merchant_agpv AS (
    SELECT m.business_uuid, m.kyc_date, m.kyc_month,
        CASE
            WHEN c.signed_agpv < 50000                                THEN '1. Under $50K'
            WHEN c.signed_agpv >= 50000  AND c.signed_agpv < 100000  THEN '2. $50K–$100K'
            WHEN c.signed_agpv >= 100000 AND c.signed_agpv < 250000  THEN '3. $100K–$250K'
            WHEN c.signed_agpv >= 250000 AND c.signed_agpv < 500000  THEN '4. $250K–$500K'
            WHEN c.signed_agpv >= 500000 AND c.signed_agpv < 1000000 THEN '5. $500K–$1M'
            WHEN c.signed_agpv >= 1000000                            THEN '6. Over $1M'
            ELSE 'Unknown'
        END AS signed_agpv_band
    FROM merchant_kyc m
    LEFT JOIN campaigns c ON m.business_uuid = c.gd_merchant_id
),
txn AS (
    SELECT m.business_uuid, m.kyc_month, m.signed_agpv_band,
        DATEDIFF('month', m.kyc_month, DATE_TRUNC('month', t.create_utc_ts)) AS months_since_kyc
    FROM merchant_agpv m
    JOIN dev.ba_commerce.txn_w_catalog t ON t.business_uuid = m.business_uuid
    WHERE t.catalog_item_flag = 1
      AND t.create_utc_ts >= m.kyc_date
      AND t.create_utc_ts >= '2025-07-01'
    GROUP BY 1, 2, 3, 4
),
cohort AS (
    SELECT kyc_month, signed_agpv_band, COUNT(DISTINCT business_uuid) AS total
    FROM txn GROUP BY 1, 2
),
ret AS (
    SELECT kyc_month, signed_agpv_band, months_since_kyc, COUNT(DISTINCT business_uuid) AS active
    FROM txn WHERE months_since_kyc >= 0 GROUP BY 1, 2, 3
)
SELECT
    TO_CHAR(r.kyc_month, 'YYYY-MM') AS kyc_month,
    r.signed_agpv_band,
    r.months_since_kyc,
    c.total AS cohort_size,
    ROUND(100.0 * r.active / NULLIF(c.total, 0), 2) AS retention_pct
FROM ret r
JOIN cohort c ON r.kyc_month = c.kyc_month AND r.signed_agpv_band = c.signed_agpv_band
ORDER BY 2, 1, 3;
